function langSelect() { var $langBtn = $('.CO05_footer-lang-select'); var $langOption = $('.option-btn__lang > a'); $langBtn.on('click', function () { $(this).toggleClass('active').siblings('.CO05_footer-lang-option').toggleClass('active'); $langOption.each(function () { $(this).on('click', function () { $(this) .parent() .addClass('selected') .siblings() .removeClass('selected') .parent($langBtn) .removeClass('active') .siblings($langBtn) .removeClass('active'); var $text = $('.option-btn__lang.selected a span').html(); $langBtn.html($text); }); }); }); }
